```sql
create table pattern_catalog (
  id serial primary key,

  pattern_id integer references patterns(id),
  name text not null,
  name_ru text not null,
  type text not null check (type in ('intraday', 'multi_day')),

  timeframe text not null, -- '15m', 'daily'

  -- Plain-language description
  description_ru text not null, -- how pattern works, explained simply

  -- Step-by-step phases (JSONB array)
  -- [{time: "9:30-10:30", name: "Дип", color: "red", description: "...", avg_move: -0.82, example: "..."}]

  -- Trigger conditions (JSONB array)
  -- [{icon: "📉", text: "Дип ≥ 0.5% в первые 60 мин"}]

  -- Core stats
  win_rate numeric(5,4) not null,
  avg_return numeric(8,4) not null,
  avg_loss numeric(8,4),
  expected_value numeric(8,4),
  risk_reward numeric(5,2),
  sample_size integer not null,
  sharpe numeric(5,2),
  confidence_grade text, -- 'A+', 'A', 'B+', 'B', 'C'

  quarter_breakdown jsonb default '{}',
  -- {monday: {win: 0.82, n: 11}, tuesday: {...}, ...}
  weekday_breakdown jsonb default '{}',

  -- Fail analysis (JSONB array)

  avg_profile real[] not null default '{}',

  -- Metadata
  active boolean default true,
  created_at timestamptz default now(),
  updated_at timestamptz default now(),

  unique(stock_id, name)
);

create index idx_pattern_catalog_stock on pattern_catalog(stock_id);
```

- [ ] **Step 3: Write builder that computes breakdowns from pattern_events**

```typescript
// src/catalog/builder.ts
import { supabase } from '../db/client.js';
import { PATTERN_TEMPLATES } from './templates.js';
import type { CatalogEntry, BreakdownEntry } from './types.js';

// Compute confidence grade from win_rate + sample_size
export function confidenceGrade(winRate: number, n: number, sharpe: number): string {
  const score = winRate * 0.4 + Math.min(n / 100, 1) * 0.3 + Math.min(sharpe / 2, 1) * 0.3;
  if (score >= 0.85) return 'A+';
  if (score >= 0.75) return 'A';
  if (score >= 0.65) return 'B+';
  if (score >= 0.55) return 'B';
  return 'C';
}

// Compute breakdown from events
export function computeBreakdown(
  events: Array<{ date: string; was_correct: boolean; profit_pct: number }>,
  keyFn: (date: string) => string
): Record<string, BreakdownEntry> {
  const groups: Record<string, { wins: number; total: number; returns: number[] }> = {};

  for (const e of events) {
    const key = keyFn(e.date);
    if (!groups[key]) groups[key] = { wins: 0, total: 0, returns: [] };
    groups[key].total++;
    if (e.was_correct) groups[key].wins++;
    groups[key].returns.push(e.profit_pct);
  }

  const result: Record<string, BreakdownEntry> = {};
  for (const [key, g] of Object.entries(groups)) {
    result[key] = {
      win: g.total > 0 ? g.wins / g.total : 0,
      n: g.total,
      avg_return: g.returns.length > 0
        ? g.returns.reduce((a, b) => a + b, 0) / g.returns.length
        : 0,
    };
  }
  return result;
}

// Get quarter label from date string
export function getQuarter(dateStr: string): string {
  const d = new Date(dateStr);
  const q = Math.ceil((d.getMonth() + 1) / 3);
  return `Q${q}_${d.getFullYear()}`;
}

// Get weekday name from date string
const WEEKDAYS = ['sunday', 'monday', 'tuesday', 'wednesday', 'thursday', 'friday', 'saturday'];
export function getWeekday(dateStr: string): string {
  return WEEKDAYS[new Date(dateStr).getDay()];
}

// Build avg profile from day_profiles where pattern triggered
export async function buildAvgProfile(
  stockId: number,
  eventDates: string[]
): Promise<number[]> {
  if (eventDates.length === 0) return [];

  const { data } = await supabase
    .from('day_profiles')
    .select('profile_vector')
    .eq('stock_id', stockId)
    .in('date', eventDates);

  if (!data || data.length === 0) return [];

  // Parse vectors and average them
  const vectors = data
    .map(d => {
      const str = d.profile_vector as string;
      if (!str) return null;
      return str.replace(/[\[\]]/g, '').split(',').map(Number);
    })
    .filter((v): v is number[] => v !== null && v.length > 0);

  if (vectors.length === 0) return [];

  const len = vectors[0].length;
  const avg = new Array(len).fill(0);
  for (const v of vectors) {
    for (let i = 0; i < len && i < v.length; i++) {
      avg[i] += v[i];
    }
  }
  return avg.map(x => Number((x / vectors.length).toFixed(4)));
}

export async function getEarningsPhase(

  dateStr: string
): Promise<string> {
  const { data } = await supabase
    .from('day_profiles')
    .select('days_since_earnings, days_until_earnings')
    .eq('stock_id', stockId)
    .eq('date', dateStr)
    .single();

  if (!data) return 'unknown';
  const since = data.days_since_earnings as number | null;
  const until = data.days_until_earnings as number | null;

  if (since !== null && since <= 5) return 'post_earnings';
  if (until !== null && until <= 10) return 'pre_earnings';
  return 'mid_quarter';
}

// Main: build catalog entries from existing patterns
export async function buildCatalog(stockId: number): Promise<CatalogEntry[]> {
  // Fetch all active/live patterns
  const { data: patterns } = await supabase
    .from('patterns')
    .select('*')
    .eq('stock_id', stockId)
    .in('lifecycle_stage', ['live', 'validated', 'monitored']);

  if (!patterns) return [];

  const entries: CatalogEntry[] = [];

  for (const pat of patterns) {
    // Find matching template
    const templateKey = matchTemplate(pat);
    const template = templateKey ? PATTERN_TEMPLATES[templateKey] : null;

    // Fetch events for breakdowns
    const { data: events } = await supabase
      .from('pattern_events')
      .select('date, was_correct, profit_pct')
      .eq('pattern_id', pat.id);

    const evts = events || [];
    const eventDates = evts.map(e => e.date);

    // Compute breakdowns
    const earningsPhases: Record<string, string> = {};
    for (const date of eventDates) {
      earningsPhases[date] = await getEarningsPhase(stockId, date);
    }

    const phaseBreakdown = computeBreakdown(evts, d => earningsPhases[d] || 'unknown');
    const quarterBreakdown = computeBreakdown(evts, getQuarter);
    const weekdayBreakdown = computeBreakdown(evts, getWeekday);

    // Build avg profile
    const winDates = evts.filter(e => e.was_correct).map(e => e.date);
    const avgProfile = await buildAvgProfile(stockId, winDates);

    // Compute sharpe (simplified)
    const returns = evts.map(e => e.profit_pct);
    const mean = returns.length > 0 ? returns.reduce((a, b) => a + b, 0) / returns.length : 0;
    const std = returns.length > 1
      ? Math.sqrt(returns.reduce((s, r) => s + (r - mean) ** 2, 0) / (returns.length - 1))
      : 1;
    const sharpe = std > 0 ? mean / std : 0;

    entries.push({
      stock_id: stockId,
      pattern_id: pat.id,
      name: template?.name || pat.type,
      name_ru: template?.name_ru || pat.description || pat.type,
      type: template?.type || 'intraday',
      source: pat.source === 'user' ? 'user' : 'system',
      timeframe: template?.timeframe || '15m',
      description_ru: template?.description_ru || pat.description || '',
      phases: template?.phases || [],
      conditions: template?.conditions || [],
      win_rate: pat.win_rate,
      avg_return: pat.avg_win || mean,
      avg_loss: pat.avg_loss || 0,
      expected_value: pat.expected_value || 0,
      risk_reward: template ? Math.abs(template.tp1_pct / template.stop_pct) : 1,
      sample_size: pat.occurrences || evts.length,
      sharpe: Number(sharpe.toFixed(2)),
      confidence_grade: confidenceGrade(pat.win_rate, pat.occurrences || evts.length, sharpe),
      phase_breakdown: phaseBreakdown,
      quarter_breakdown: quarterBreakdown,
      weekday_breakdown: weekdayBreakdown,
      fail_reasons: template?.fail_reasons || [],
      entry_rule: template?.entry_rule || '',
      entry_time: template?.entry_time || '',
      stop_pct: template?.stop_pct || -1.0,
      tp1_pct: template?.tp1_pct || 1.5,
      tp2_pct: template?.tp2_pct || 2.5,
      avg_profile: avgProfile,
    });
  }

  return entries;
}

// Match pattern DB record to a template key
function matchTemplate(pattern: any): string | null {
  const type = pattern.type as string;
  const params = pattern.parameters as any;

  if (type === 'mean_reversion' && params?.direction === 'down' && params?.threshold <= 1.5) {
    return 'mean_reversion:down:0.5-1.5';
  }
  if (type === 'gap_fill' && params?.direction === 'up') {
    return 'gap_fill:up';
  }
  if (type === 'volume_surge' && params?.direction === 'up') {
    return 'volume_surge:up';
  }
  // Multi-day patterns
  if (type === 'mean_reversion' && params?.observation_window >= 20) {
    return 'mean_reversion:multi_day';
  }
  if (type === 'post_earnings_drift' || (type === 'gap_fill' && params?.is_earnings)) {
    return 'post_earnings_drift';
  }

  return null;
}

// Store catalog entries to DB
export async function storeCatalog(entries: CatalogEntry[]): Promise<void> {
  for (const entry of entries) {
    const { error } = await supabase
      .from('pattern_catalog')
      .upsert({
        ...entry,
        updated_at: new Date().toISOString(),
      }, { onConflict: 'stock_id,name' });

    if (error) console.error(`Failed to store catalog entry ${entry.name}:`, error.message);
  }
}
```

```typescript
// Compute trade levels from open price and catalog percentages
export function computeTradeLevels(
  openPrice: number,
  stopPct: number,
  tp1Pct: number,
  tp2Pct: number
): { entry: number; stop: number; tp1: number; tp2: number } {
  return {
    entry: openPrice,
    stop: Number((openPrice * (1 + stopPct / 100)).toFixed(2)),
    tp1: Number((openPrice * (1 + tp1Pct / 100)).toFixed(2)),
    tp2: Number((openPrice * (1 + tp2Pct / 100)).toFixed(2)),
  };
}

// Determine current phase based on time elapsed and price action
export function determinePhase(
  phases: Array<{ name: string; time: string }>,
  minutesSinceOpen: number
): { current: string; progress: Array<{ name: string; status: 'done' | 'active' | 'pending' }> } {
  // Map phase time ranges to minutes
  const phaseMinutes = [60, 150, 240, 390]; // approximate boundaries

  let currentIdx = 0;
  for (let i = 0; i < phaseMinutes.length; i++) {
    if (minutesSinceOpen < phaseMinutes[i]) {
      currentIdx = i;
      break;
    }
    if (i === phaseMinutes.length - 1) currentIdx = i;
  }

  return {
    current: phases[currentIdx]?.name || 'unknown',
    progress: phases.map((p, i) => ({
      name: p.name,
      status: i < currentIdx ? 'done' as const : i === currentIdx ? 'active' as const : 'pending' as const,
    })),
  };
}

// Generate tags from market context
export function generateTags(
  spyPct: number,
  volumeRatio: number,
  earningsPhase: string
): string[] {
  const tags: string[] = [];

  if (spyPct > 0.3) tags.push('bullish');
  else if (spyPct < -0.3) tags.push('bearish');
  else tags.push('neutral');

  if (volumeRatio > 1.3) tags.push(`vol_${volumeRatio.toFixed(1)}x`);
  tags.push(earningsPhase.replace('_', '-'));

  return tags;
}

// Main: generate signals for today
export async function generateSignals(
  stockId: number,
  ticker: string
): Promise<LiveSignal[]> {
  // Get catalog entries
  const { data: catalog } = await supabase
    .from('pattern_catalog')
    .select('*')
    .eq('stock_id', stockId)
    .eq('active', true);

  if (!catalog || catalog.length === 0) return [];

  // Get current session data
  const analogResult = await analyzeCurrentSession(stockId, ticker);
  if (!analogResult) return [];

  const currentPct = analogResult.current_pct;
  const candlesSoFar = analogResult.candles_so_far;
  const minutesSinceOpen = candlesSoFar * 15;

  // Get today's open price
  const today = toETDate(Date.now());
  const { data: todayCandle } = await supabase
    .from('candles_daily')
    .select('open, high, low, close')
    .eq('stock_id', stockId)
    .eq('date', today)
    .single();

  const openPrice = todayCandle?.open || 0;
  const currentPrice = todayCandle?.close || openPrice * (1 + currentPct / 100);

  // Get active pattern signals from composite scoring
  const patternSignals = await getActivePatternSignals(stockId, []);

  const signals: LiveSignal[] = [];

  for (const cat of catalog) {
    // Simple match scoring based on current price action vs avg_profile
    const avgProfile = cat.avg_profile as number[] || [];
    if (avgProfile.length === 0) continue;

    const profileSlice = avgProfile.slice(0, candlesSoFar);
    if (profileSlice.length === 0) continue;

    // Compute similarity between current intraday path and pattern's avg profile
    let matchScore = 0;
    if (analogResult.analogs && analogResult.analogs.length > 0) {
      // Use analog similarity as proxy for pattern match
      const topSimilarity = analogResult.analogs[0].similarity;
      matchScore = topSimilarity * 100;
    } else {
      // Fallback: compare direction
      const patternDirection = profileSlice[profileSlice.length - 1] > 0 ? 'up' : 'down';
      const currentDirection = currentPct > 0 ? 'up' : 'down';
      matchScore = patternDirection === currentDirection ? 60 : 30;
    }

    if (matchScore < 40) continue; // skip weak matches

    const phases = cat.phases as Array<{ name: string; time: string }> || [];
    const { current, progress } = determinePhase(phases, minutesSinceOpen);

    const levels = computeTradeLevels(
      openPrice,
      cat.stop_pct as number || -1,
      cat.tp1_pct as number || 2,
      cat.tp2_pct as number || 3
    );

    signals.push({
      catalog_id: cat.id,
      pattern_name: cat.name as string,
      pattern_name_ru: cat.name_ru as string,
      match_pct: Number(matchScore.toFixed(1)),
      current_phase: current,
      phase_progress: progress,
      open_price: openPrice,
      entry_price: levels.entry,
      stop_price: levels.stop,
      tp1_price: levels.tp1,
      tp2_price: levels.tp2,
      current_price: Number(currentPrice.toFixed(2)),
      market_context: { spy_pct: 0, qqq_pct: 0 }, // filled by caller
      tags: [],
      analysis_text: '',
    });
  }

  // Sort by match %
  signals.sort((a, b) => b.match_pct - a.match_pct);

  return signals;
}
```

```typescript
// supabase/functions/api/index.ts
import { createClient } from 'https://esm.sh/@supabase/supabase-js@2';

const corsHeaders = {
  'Access-Control-Allow-Origin': '*',
  'Access-Control-Allow-Headers': 'authorization, x-client-info, apikey, content-type',

Deno.serve(async (req) => {
  if (req.method === 'OPTIONS') {
    return new Response(null, { headers: corsHeaders });
  }

  const url = new URL(req.url);
  const path = url.pathname.replace('/api', '');
  const supabase = createClient(
    Deno.env.get('SUPABASE_URL')!,
    Deno.env.get('SUPABASE_SERVICE_ROLE_KEY')!
  );

  try {
    // GET /catalog?ticker=ALAB
    if (path === '/catalog' && req.method === 'GET') {
      const ticker = url.searchParams.get('ticker') || 'ALAB';
      const { data: stock } = await supabase
        .from('stocks').select('id').eq('ticker', ticker).single();
      if (!stock) return json({ error: 'Stock not found' }, 404);

      const { data } = await supabase
        .from('pattern_catalog')
        .select('*')
        .eq('stock_id', stock.id)
        .eq('active', true)
        .order('win_rate', { ascending: false });

      return json(data || []);
    }

    // GET /catalog/:id
    if (path.startsWith('/catalog/') && req.method === 'GET') {
      const id = parseInt(path.split('/')[2]);
      const { data } = await supabase
        .from('pattern_catalog').select('*').eq('id', id).single();
      return json(data);
    }

    // GET /signals?ticker=ALAB
    if (path === '/signals' && req.method === 'GET') {
      const ticker = url.searchParams.get('ticker') || 'ALAB';
      const { data: stock } = await supabase
        .from('stocks').select('id').eq('ticker', ticker).single();
      if (!stock) return json({ error: 'Stock not found' }, 404);

      const today = new Date().toISOString().split('T')[0];
      const { data } = await supabase
        .from('pattern_signals')
        .select('*, pattern_catalog!inner(name, name_ru, phases, stop_pct, tp1_pct, tp2_pct, avg_profile)')
        .eq('stock_id', stock.id)
        .eq('date', today)
        .order('match_pct', { ascending: false });

      return json(data || []);
    }

    // GET /history?ticker=ALAB&from=2026-03-01&to=2026-03-17
    if (path === '/history' && req.method === 'GET') {
      const ticker = url.searchParams.get('ticker') || 'ALAB';
      const from = url.searchParams.get('from');
      const to = url.searchParams.get('to');

      const { data: stock } = await supabase
        .from('stocks').select('id').eq('ticker', ticker).single();
      if (!stock) return json({ error: 'Stock not found' }, 404);

      let query = supabase
        .from('pattern_signals')
        .select('date, match_pct, current_phase, actual_return, was_correct, pattern_catalog!inner(name_ru)')
        .eq('stock_id', stock.id)
        .order('date', { ascending: false });

      if (from) query = query.gte('date', from);
      if (to) query = query.lte('date', to);

      const { data } = await query.limit(60);
      return json(data || []);
    }

    // GET /price?ticker=ALAB
    if (path === '/price' && req.method === 'GET') {
      const ticker = url.searchParams.get('ticker') || 'ALAB';
      const { data: stock } = await supabase
        .from('stocks').select('id').eq('ticker', ticker).single();
      if (!stock) return json({ error: 'Stock not found' }, 404);

      const { data: daily } = await supabase
        .from('candles_daily')
        .select('date, open, high, low, close, volume, gap_pct')
        .eq('stock_id', stock.id)

        .limit(1)
        .single();

        .from('day_profiles')
        .select('day_change_pct, days_until_earnings, days_since_earnings, quarter_position')

        .order('date', { ascending: false })
        .limit(1)
        .single();

      return json({ daily, profile, ticker });
    }

    // GET /stocks
    if (path === '/stocks' && req.method === 'GET') {
      const { data } = await supabase
        .from('stocks').select('id, ticker, name, sector').eq('active', true);
      return json(data || []);
    }

    return json({ error: 'Not found' }, 404);
  } catch (err) {
    return json({ error: (err as Error).message }, 500);
  }
});

function json(data: any, status = 200) {
  return new Response(JSON.stringify(data), {
    status,
    headers: { ...corsHeaders, 'Content-Type': 'application/json' },
  });
}
```

```typescript
const API_URL = import.meta.env.VITE_API_URL || 'http://localhost:54321/functions/v1/api';

async function fetchApi<T>(path: string, params?: Record<string, string>): Promise<T> {
  const url = new URL(`${API_URL}${path}`);
  if (params) Object.entries(params).forEach(([k, v]) => url.searchParams.set(k, v));
  const res = await fetch(url.toString());
  if (!res.ok) throw new Error(`API error: ${res.status}`);
  return res.json();
}

export const api = {
  getCatalog: (ticker: string) => fetchApi<any[]>('/catalog', { ticker }),
  getCatalogItem: (id: number) => fetchApi<any>(`/catalog/${id}`),
  getSignals: (ticker: string) => fetchApi<any[]>('/signals', { ticker }),
```
